	/**
	 * Check titles.
	 *
	 * @param title the title
	 * @return true, if successful
	 */
	public String checkTitles(String title) {
		try {
			if (title == null || title.isEmpty()) {
				return null;
			}
				
			for (String Title : titles) {
				if (title.equalsIgnoreCase(Title)) {
					return title;
				}
			}
		} catch (Exception e) {
			LOGGER.trace(ERROR, e);
		}
		return null;
	}
